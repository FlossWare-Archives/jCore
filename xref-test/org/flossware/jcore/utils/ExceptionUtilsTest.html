17  package org.flossware.jcore.utils;
18  
19  import java.io.IOException;
20  import java.lang.reflect.Constructor;
21  import java.lang.reflect.InvocationTargetException;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  /**
26   * Tests the ExceptionUtils utility class.
27   *
28   * @author Scot P. Floess
29   */
30  public class ExceptionUtilsTest {
31  
32      class TestException extends Exception {
33  
34          public TestException() {
35  
36          }
37  
38          public TestException(final Throwable causedBy) {
39              super(causedBy);
40          }
41      }
42  
43      /**
44       * Tests the constructor.
45       */
46      @Test
47      public void testConstructor() throws NoSuchMethodException, InstantiationException, IllegalAccessException, IllegalArgumentException, InvocationTargetException {
48          final Constructor constructor = ExceptionUtils.class.getDeclaredConstructor();
49          constructor.setAccessible(true);
50          constructor.newInstance(new Object[0]);
51      }
52  
53      /**
54       * Tests of containsThrowable is processable.
55       */
56      @Test
57      public void test_isContainsProcessable() {
58          Assert.assertFalse("Should not be processable", ExceptionUtils.isContainsProcessable(null, null));
59          Assert.assertFalse("Should not be processable", ExceptionUtils.isContainsProcessable(null, ExceptionUtilsTest.class));
60          Assert.assertFalse("Should not be processable", ExceptionUtils.isContainsProcessable(new RuntimeException(), null));
61          Assert.assertTrue("Should be processable", ExceptionUtils.isContainsProcessable(new NullPointerException(), RuntimeException.class));
62      }
63  
64      /**
65       * Tests containing a throwable via its class.
66       */
67      @Test
68      public void test_contains_class() {
69          Assert.assertFalse("Should not contain the exception", ExceptionUtils.containsThrowable(new RuntimeException(), NullPointerException.class));
70          Assert.assertFalse("Should not contain the exception", ExceptionUtils.containsThrowable(new RuntimeException(new IllegalArgumentException()), NullPointerException.class));
71          Assert.assertTrue("Should contain the exception", ExceptionUtils.containsThrowable(new RuntimeException(), RuntimeException.class));
72          Assert.assertTrue("Should contain the exception", ExceptionUtils.containsThrowable(new NullPointerException(), RuntimeException.class));
73          Assert.assertTrue("Should contain the exception", ExceptionUtils.containsThrowable(new TestException(new IOException()), TestException.class));
74          Assert.assertTrue("Should contain the exception", ExceptionUtils.containsThrowable(new TestException(new TestException(new IOException())), IOException.class));
75      }
76  
77      /**
78       * Tests containing a throwable via instance.
79       */
80      @Test(expected = IllegalArgumentException.class)
81      public void test_contains_throwable_null() {
82          ExceptionUtils.containsThrowable(new RuntimeException(), (Throwable) null);
83      }
84  
85      /**
86       * Tests containing a throwable via instance.
87       */
88      @Test
89      public void test_contains_throwable() {
90          Assert.assertFalse("Should not contain the exception", ExceptionUtils.containsThrowable(new RuntimeException(), new NullPointerException()));
91          Assert.assertFalse("Should not contain the exception", ExceptionUtils.containsThrowable(new RuntimeException(new IllegalArgumentException()), new NullPointerException()));
92          Assert.assertTrue("Should contain the exception", ExceptionUtils.containsThrowable(new RuntimeException(), new RuntimeException()));
93          Assert.assertTrue("Should contain the exception", ExceptionUtils.containsThrowable(new NullPointerException(), new RuntimeException()));
94          Assert.assertTrue("Should contain the exception", ExceptionUtils.containsThrowable(new TestException(new IOException()), new TestException()));
95          Assert.assertTrue("Should contain the exception", ExceptionUtils.containsThrowable(new TestException(new TestException(new IOException())), new IOException()));
96      }
97  
98      /**
99       * Tests if a null exception message is contained.
100      */
101     @Test(expected = IllegalArgumentException.class)
102     public void test_isExceptionMsgContained_msg_null() {
103         ExceptionUtils.isExceptionMsgContained(new Throwable(), null);
104     }
105 
106     /**
107      * Tests if a blank exception message is contained.
108      */
109     @Test(expected = IllegalArgumentException.class)
110     public void test_isExceptionMsgContained_msg_blank() {
111         ExceptionUtils.isExceptionMsgContained(new Throwable(), "    ");
112     }
113  
114     /**
115      * Tests if an empty exception message is contained.
116      */
117     @Test(expected = IllegalArgumentException.class)
118     public void test_isExceptionMsgContained_msg_empty() {
119         ExceptionUtils.isExceptionMsgContained(new Throwable(), "");
120     }
121     
122     /**
123      * Tests if an exception message is contained.
124      */
125     @Test
126     public void test_isExceptionMsgContained() {
127         Assert.assertFalse("Should not contain the message", ExceptionUtils.isExceptionMsgContained(new NullPointerException("alpha"), "foo"));
128         Assert.assertTrue("Should contain the message", ExceptionUtils.isExceptionMsgContained(new NullPointerException("alpha"), "alpha"));
129         
130         final InvocationTargetException ite = new InvocationTargetException(new RuntimeException("beta"), "alpha");
131         Assert.assertFalse("Should not contain the message", ExceptionUtils.isExceptionMsgContained(ite, "alpha"));
132         Assert.assertTrue("Should contain the message", ExceptionUtils.isExceptionMsgContained(ite, "beta"));
133     }
134 }
