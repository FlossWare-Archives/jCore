3   import org.flossware.jcore.utils.StringUtils;
4   
5   /**
6    *
7    * Default abstract base class of stringifiables.
8    *
9    * @author sfloess
10   *
11   */
12  public abstract class AbstractStringifiable extends AbstractCommonBase implements Stringifiable {
13      /**
14       * Default constructor.
15       */
16      protected AbstractStringifiable() {
17      }
18  
19      protected StringBuilder appendLine(final StringBuilder stringBuilder, final String... line) {
20          return StringUtils.concat(stringBuilder, line).append(LINE_SEPARATOR_STRING);
21      }
22  
23      /**
24       * {@inheritDoc}
25       */
26      @Override
27      public StringBuilder toStringBuilder(final StringBuilder stringBuilder) {
28          return toStringBuilder(stringBuilder, "");
29      }
30  
31      /**
32       * {@inheritDoc}
33       */
34      @Override
35      public StringBuilder toStringBuilder(final String prefix) {
36          return toStringBuilder(new StringBuilder(), prefix);
37      }
38  
39      /**
40       * {@inheritDoc}
41       *
42       * @return the string representation of self.
43       */
44      @Override
45      public String toString() {
46          return toStringBuilder("").toString();
47      }
48  }
